cmake_minimum_required(VERSION 3.7)

project(iso-x86-bios)
message(STATUS "Project " ${PROJECT_NAME})

add_custom_command(OUTPUT "${HHUOS_ROOT_DIR}/hhuOS.iso"
        COMMAND grub-mkrescue /usr/lib/grub/i386-pc -o ${HHUOS_ROOT_DIR}/hhuOS.iso ${HHUOS_ROOT_DIR}/loader/
        DEPENDS system-x86-bios)

add_custom_target(iso-x86-bios DEPENDS "${HHUOS_ROOT_DIR}/hhuOS.iso")
