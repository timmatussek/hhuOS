cmake_minimum_required(VERSION 3.7)

project(system-x86-bios)
message(STATUS "Project " ${PROJECT_NAME})

include_directories(${HHUOS_SRC_DIR})
include_directories(${X86_SRC_DIR})
include_directories(${X86_BIOS_SRC_DIR})

# Set source files
set(SOURCE_FILES
        ${X86_BIOS_SRC_DIR}/boot.asm
        ${X86_SRC_DIR}/asm_interface.cpp
        ${HHUOS_SRC_DIR}/BuildConfig.cpp
        ${HHUOS_SRC_DIR}/GatesOfHell.cpp
        ${X86_SRC_DIR}/X86Platform.cpp
        ${X86_BIOS_SRC_DIR}/X86BiosPlatform.cpp)

add_executable(${PROJECT_NAME} ${SOURCE_FILES})

set_target_properties(${PROJECT_NAME} PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${HHUOS_ROOT_DIR}/loader/boot/)
set_target_properties(${PROJECT_NAME} PROPERTIES OUTPUT_NAME "hhuOS.bin")

target_link_libraries(${PROJECT_NAME} kernel-x86 kernel device device-x86 device-x86-bios util)
